// 1. 两数之和 用一个map记录遍历过的对象，键名为数组的值，键值为数组的索引，如果map.has判断有这个值，则返回结果，若没有则记录map.set(nums[i], i)，循环最后返回[]
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    let map = new Map();
    for(let i = 0, len = nums.length; i < len; i++) {
        let searchNum = target - nums[i];
        if (map.has(searchNum)) {
            return [map.get(searchNum), i]
        } else {
            map.set(nums[i], i)
        }
    }
    return []
};

//  128. 最长连续序列：先遍历一次数组，用set接收结果，接着遍历set，如果当前是左边界，则循环判断当前值+1是否存在，直至不存在为止，此时应为当前值的最长序列长度，与记录的longestLength对比，最终得到结果。
// 注意：每次先判断!set.has(i - 1)，再设置两个变量，一个是记录值，一个记录长度
/**
 * @param {number[]} nums
 * @return {number}
 */
var longestConsecutive = function(nums) {
    let set = new Set()
    for(let i = 0, len = nums.length; i < len; i++) {
        set.add(nums[i])
    }
    let longestLength = 0;
    for(let i of set) {
        if (!set.has(i - 1)) {
            let currentNum = i,
                currentLenght = 1;
            while(set.has(currentNum + 1)) {
                currentNum++;
                currentLenght++
            }
            longestLength = Math.max(currentLenght, longestLength)
        }
    }
    return longestLength
};

// 11. 盛最多水的容器：用左右指针，每次移动算结果(Math.min(height[left], height[right])) * (right - left)，然后与之前的最大值对比，选择其中最大的，然后如果height[right] > height[left]，说明left需要往前走，尽量保证水柱是最高的
// 注意边界r = length - 1
/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function(height) {
    if (!height || !height.length) return;
    let l = 0, r = height.length - 1;
    let maxValue = 0;
    while(l < r) {
        const value = (Math.min(height[l], height[r])) * (r - l)
        maxValue = Math.max(value, maxValue)
        height[l] > height[r] ? r-- : l++
    }
    return maxValue
};


// 15. 三数之和
// 1.判断边界条件，小于三个数返回；2.排序 3.使用左右指针，判断当前sum大于0说明选的右值太大，需要-1，小于0说明左值太小，需要+1 4.考虑优化，如果当前值大于0，说明后面的值都会大于0，直接break
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var threeSum = function(nums) {
    if (!nums || !nums.length) return [];
    nums.sort((a, b) => a - b)
    let res = [];
    for(let i = 0, n = nums.length; i < n; i++) {
        if (nums[i] > 0) break;
        if (i > 0 && nums[i] === nums[i - 1]) continue;
        let l = i + 1, r = n - 1;
        while(l < r) {
            const sum = nums[i] + nums[l] + nums[r]
            if (sum === 0) {
                res.push([nums[i], nums[l], nums[r]])
                while(l < r && nums[l] === nums[l + 1]) l++
                while(l < r && nums[r] === nums[r - 1]) r--
                l++
                r--
            } else if (sum > 0) r--;
            else l++
        }
    }
    return res;
};

// 3. 无重复字符的最长子串：用一个map接收，建名为字符，键值为索引，当当前值在map中有时，当前索引-取出map的索引（j-i）并且更新当前值的索引，如果没有，则当前索引（j-i+1）用一个最大值始终对比保存，最终返回，总结两种情况i = Math.max(map.get(s[j]) + 1, i)，longestLen = Math.max(j - i + 1, longestLen)
<!-- 为了保持窗口内无重复字符，左指针 i 必须移动到“重复字符上一次出现位置”的‌下一位‌。 -->
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function(s) {
    const map = new Map();
    let longestLen = 0;
    for(let i = 0, j = 0, n = s.length; j < n; j++) {
        if (map.has(s[j])) {
            i = Math.max(map.get(s[j]) + 1, i)
        }
        longestLen = Math.max(j - i + 1, longestLen)
        map.set(s[j], j)
    }
    return longestLen
};

 /*
 1. 维护一个双端队列 queue，使得队列中的索引对应的 nums 值是 单调递减 的。
 2. 每次滑动窗口时：
 3. 先移除队列中不在窗口范围内 的元素（i - k) 之前的。
 4. 再从队列尾部移除比当前 nums[i] 小的元素，保证队列单调递减。
 5. 队列头部元素始终是窗口内的最大值，存入 ans[]。

/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var maxSlidingWindow = function(nums, k) {
    let ans = [],
        queue = []
    for(let i = 0, n = nums.length; i < n; i++) {
        if (queue.length > 0 && queue[0] < i - k + 1) {
            queue.shift()
            <!-- 首先要剔除掉多余窗口的元素 -->
        }
        while(queue.length > 0 && nums[queue[queue.length - 1]] < nums[i]) {
            queue.pop()
            <!-- 因为 nums[i] 不仅值更大，而且它在数组中的位置更靠后（索引更大），这意味着 nums[i] 会比队列尾部的那个元素更晚离开滑动窗口。既然有一个更大且更持久的候选者存在，那个较小的旧元素就失去了竞争最大值的资格。 -->
        }
        queue.push(i);
        if (i >= k - 1) {
            ans.push(nums[queue[0]])
        }
    }
    return ans
};

/**

 * @return {number[][]}
 */
var merge = function(intervals) {
    intervals.sort((a, b) => a[0] - b[0])
    let ans = []
    for(let i = 0, n = intervals.length; i < n; i++) {
        if (ans.length && ans[ans.length - 1][1] >= intervals[i][0]) {

            ans.push(intervals[i])
        }
    }
    return ans;
};

// 238. 除自身以外数组的乘积：
// 方法1:
// 前一个乘积 pre[i] = pre[i - 1] * nums[i - 1] pre[0] = 1
// 后一个乘积 suf[i] = suf[i + 1] * nums[i + 1] suf[nums.length - 1] = 1
// 最终 res[i] = pre[i] * suf[i]
// 方法2:
// 先设置一个suf，suf[i]*=pre，用变量pre储存pre的乘积
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var productExceptSelf = function(nums) {
    let n = nums.length;
    let suf = new Array(n).fill(1);
    for(let i = n - 2; i >= 0; i--) {
        suf[i] = suf[i + 1] * nums[i + 1]
    }
    let prev = 1;
    for(let i = 0; i < n; i++) {
        suf[i] *= prev
        prev *= nums[i]
    }
    return suf
};

 * Definition for a binary tree node.

 *     this.right = (right===undefined ? null : right)

 * @param {number[]} preorder

 * @return {TreeNode}
 */
var buildTree = function(preorder, inorder) {
    if (!preorder.length || !inorder.length) return null;
    let rootVal = preorder[0];
    let mid = inorder.indexOf(rootVal);
    let root = new TreeNode(rootVal);
    root.left = buildTree(preorder.slice(1, mid + 1), inorder.slice(0, mid))
    root.right = buildTree(preorder.slice(mid + 1), inorder.slice(mid + 1))
    return root
};

    let map = new Map();

    let ans = 0;
    map.set(0, 1)
    var dfs = function(root, sum) {
        if (!root) return;
        sum += root.val
        ans += (map.get(sum - targetSum) ?? 0);
        map.set(sum, (map.get(sum) ?? 0) + 1)

        dfs(root.right, sum)

    }
    dfs(root, sum)
    return ans
}

 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)

/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxPathSum = function(root) {
    let ans = -Infinity;
    var dfs = function(root) {
        if (!root) return 0;
        let val = root.val
        let left = dfs(root.left);
        let right = dfs(root.right);
        ans = Math.max(val + left + right, ans);
        return Math.max(val + Math.max(left, right), 0)
    }
    dfs(root)
    return ans
};

var decodeString = function(s) {
    if (!s || !s.length) return '';
    let s0 = s[0];
    if (/^[a-z]$/.test(s0)) return s0 + decodeString(s.slice(1));
    let i = s.indexOf('[');

    for(let j = i + 1; j < s.length; j++) {
        if (s[j] === '[') {
            balance++
        } else if (s[j] === ']') {
            balance--
            if (balance === 0) {
                let k = parseInt(s.slice(0, i))
                return decodeString(s.slice(i + 1, j)).repeat(k) + decodeString(s.slice(j + 1))
            }
        }
    }

// 114. 二叉树展开为链表：可以理解为左子树处理为null，右子树处理为先序遍历的结果，先处理左右子树为一个单链表，然后左节点设置为null，右节点为右节点，然后递归直至右节点为null，此时节点的右节点为此前保存的左节点
// 注意函数不返回结果，那么结果应该是在left = root.left这样，而不是函数返回的结果
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {void} Do not return anything, modify root in-place instead.
 */
var flatten = function(root) {
    if (!root) return;
    flatten(root.left)
    flatten(root.right)
    let left = root.left;
    let right = root.right;
    root.left = null;
    root.right = left;
    while(root.right !== null) {
        root = root.right
    }
    root.right = right
};



// 73. 矩阵置零：
// 第一循环先把第一行或第一列该设置为0的数设置为0
// 第二循环将0以外的行列设置为0
// 通过变量firstRowHasZero，firstColHasZero设置第一行，第一列为0
/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var setZeroes = function(matrix) {
    if (!matrix.length || !matrix[0].length) return;
    let m = matrix.length, n = matrix[0].length;
    let firstRowHasZero = matrix[0].includes(0);
    let firstColHasZero = matrix.some(item => item[0] === 0)
    for(let i = 1; i < m; i++) {
        for(let j = 1; j < n; j++) {
            if (matrix[i][j] === 0) {
                matrix[i][0] = 0;
                matrix[0][j] = 0;
            }
        }
    }
    for(let i = 1; i < m; i++) {
        for(let j = 1; j < n; j++) {
            if (matrix[i][0] === 0 || matrix[0][j] === 0) {
                matrix[i][j] = 0;
            }
        }
    }
    if (firstRowHasZero) matrix[0].fill(0)
    if (firstColHasZero) {
        for (let i of matrix) {
            i[0] = 0
        }
    }
};



// 54. 螺旋矩阵：解决问题的关键有四个变量，i,j表示在当前二维数组的位置matrix[i][j]，di表示当前取值所需转换方向，一共四个方向，所以当当前的方向不满足条件时，di = (di + 1) % 4，重新获得方向，接着再用di处理下一个i，j位置，最后一个变量则是用来存放ans中索引对应的值
// 由于旋转方向为➡️⬇️⬅️⬆️，因此数组也是[[0, 1],[1, 0], [0, -1], [-1, 0] ]
// ans[k] = matrix[i][j]
// x >= m y >= n
/**
 * @param {number[][]} matrix
 * @return {number[]}
 */
const DIRS = [[0, 1], [1, 0], [0, -1], [-1, 0]]
var spiralOrder = function(matrix) {
    let m = matrix.length, n = matrix[0].length;
    let i = 0, j = 0, di = 0;
    let ans = [];
    for(let k = 0; k < m * n; k++) {
        ans[k] = matrix[i][j];
        matrix[i][j] = Infinity;
        let x = i + DIRS[di][0];
        let y = j + DIRS[di][1];
        if (x < 0 || x >= m || y < 0 || y >= n || matrix[x][y] === Infinity) {
            di = (di + 1) % 4
        }
        i += DIRS[di][0];
        j += DIRS[di][1];
    }
    return ans;
};
