# 冒泡排序 
相邻两个元素比较大小，小的往前排
// https://blog.csdn.net/qq_48896417/article/details/138135738 j j+ 1
function bubbleSort(list) {
    for (let i = 0; i < list.length; i++) {
        for (let j = 0; j < list.length - 1 - i; j++) {
            if (list[j + 1] > list[j]) {
                let tmp = list[j + 1];
                list[j + 1] = list[j]
                list[j] = tmp
            }
        }
    }
    return list
}

# 选择排序 
选择最小的元素往前排 min i
function selectSort (list) {
    for (let i = 0; i < list.length; i++) {
        let min = i
        for (let j = i + 1; j < list.length; j++) {
            if (list[i] > list[j]) {
                min = j
            }
        }
        let tmp = list[i]
        list[i] = list[min]
        list[min] = tmp
    }
    return list
}

# 快排
function quickSort(list) {
    if (list.length <= 1) {
        return list;
    }
    let middle = Math.floor(list.length / 2);
    let middleVal = list.splice(middle, 1)[0]
    let left = [];
    let right = [];
    for (let i = 0; i < list.length; i++) {
        if (middleVal > list[i]) {
            left.push(list[i])
        } else {
            right.push(list[i])
        }
    }
    return quickSort(left).concat([middleVal], quickSort(right))
}
