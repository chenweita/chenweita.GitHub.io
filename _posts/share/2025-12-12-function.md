# debounce方法

    function deBounce(fn, delay = 1000) {
            let timer = null;
            return function (...args) {
                clearTimeout(timer);
                timer = setTimeout(() => {
                    fn.apply(this, args)
                }, delay)
            }
        }   

# deepclone方法

        function deepClone(obj) {
            if (!obj || typeof obj !== 'Object') return obj;
            let resObj = Array.isArray(obj) ? [] : {}
            for (let i in obj) {
                if (obj instanceof Object) {
                    resObj[i] = deepClone(obj[i])
                } else {
                    resObj[i] = obj[i]
                }
            }
            return resObj;
        }
        function isObj(obj) {
            return Object(obj) === obj
        }
        function deepClone(obj) {
            if (typeof obj === 'symbol') {
                return Symbol(obj.description)
            }
            if (obj instanceof Map) {
                return new Map([...obj])
            }
            if (obj instanceof Set) {
                return new Set([...obj])
            }
            if (!isObj || typeof obj === 'function') return obj;
            let newobj = Array.isArray(obj) ? [] : {}
            for (let i in obj) {
                newobj[i] = deepClone(obj[i])
            }
            let symbols = Object.getOwnPropertySymbols(obj);
            for (let i of symbols) {
                newobj[i] = deepClone(obj[i])
            }
            return newobj;
        }

# throttle
        function throttle(fn, delay = 1000) {
            let timer = null;
            return function(...args) {
                if (!timer) {
                    timer = setTimeout(() => {
                        fn.apply(this, args)
                    },delay)
                    timer = null
                }
            }
        }

# loadshGet

function loadshGet(obj, path, defaultValue) {
    if (!obj || !path) return defaultValue;
    let newPath = [];
    if (Array.isArray(path)) {
        newPath = path
    } else {
        // 不考虑[]，不然的话就直接concat把数组全部连接起来
        newPath = path.replace(/\[/g, '.').replace(/\]/g, '').split('.');
    }
    console.log(newPath)
    return newPath.reduce((obj, k) => {
        return (obj || {})[k]
    },obj) || defaultValue
}

const data = {
    user: {
      name: 'John',
      address: {
        city: 'New York'
      }
    }
  };
  
  console.log(loadshGet(data, 'user.address.city')); // 输出: New York
  console.log(loadshGet(data, '[user][address][city]')); 
  console.log(loadshGet(data, 'user.address.zip', 'Not Found')); // 输出: Not Found（zip属性不存在）

    class LRUCache {
        constructor(length) {
            this.cache = new Map();
            this.length = length;
        }
        get(key) {
            if (this.cache.has(key)) {
                let value = this.cache.get(key)
                this.cache.delete(key)
                this.cache.set(key, value)
                return value
            }
            return null
        }
        put(key, value) {
            if (this.cache.has(key)) {
                this.cache.delete(key)
            } else if (this.cache.size >= this.length) {
                let keys = this.cache.keys();
                this.cache.delete(keys.next().value)
            }
            this.cache.set(key, value)
        }
        display() {
            for (let [key, value] of this.cache) {
                console.log(key + " -> " + value);
            }
        }
    }
    const historyCache = new LRUCache(5); // 设置缓存容量为5
    
    historyCache.put("Page 1", "www.page1.com");
    historyCache.put("Page 2", "www.page2.com");
    historyCache.put("Page 3", "www.page3.com");
    historyCache.get("Page 1");
    
    // 输出浏览历史记录
    historyCache.display();

# 对比版本

unction compareVersion(v1, v2) {
    let v1List = v1.split('.').map(Number)
    let v2List = v2.split('.').map(Number)
    let length = Math.max(v1List.length, v2List.length) 
    for (let i = 0; i < length; i++) {
        if (v1List[i] > v2List[i]) {
            return 1
        }
        if (v1List[i] < v2List[i]) {
            return -1
        }
    }
    return 0
}

console.log(compareVersion("1.2.3", "1.2.3")); // 0
console.log(compareVersion("1.2.3", "1.2.4")); // -1
console.log(compareVersion("1.2.10", "1.2.2")); // 1
console.log(compareVersion("1.2.10", "1.2")); // 1
