class EventBus {
    constructor() {
        this.events = {};
        this.pendingEmits = {};
    }

    // 订阅事件
    on(event, callback) {
        if (!this.events[event]) {
            this.events[event] = [];
        }
        this.events[event].push(callback);

        // 检查是否有待触发的事件
        if (this.pendingEmits[event]) {
            this.pendingEmits[event].forEach(args => {
                callback.apply(this, args);
            });
            delete this.pendingEmits[event];
        }
    }

    // 发布事件
    emit(event, ...args) {
        if (this.events[event]) {
            this.events[event].forEach(callback => {
                callback.apply(this, args);
            });
        } else {
            // 如果没有订阅者，暂存事件
            if (!this.pendingEmits[event]) {
                this.pendingEmits[event] = [];
            }
            this.pendingEmits[event].push(args);
        }
    }

    // 取消订阅
    off(event, callback) {
        if (this.events[event]) {
            const index = this.events[event].indexOf(callback);
            if (index > -1) {
                this.events[event].splice(index, 1);
            }
        }
    }

    // 订阅一次
    once(event, callback) {
        const fn = (...args) => {
            callback.apply(this, args);
            this.off(event, fn);
        };
        this.on(event, fn);
    }
}

// 使用示例
const eventBus = new EventBus();

// 先emit事件
eventBus.emit('test', 'Hello World');
eventBus.emit('test2', { name: 'John', age: 30 });

// 后定义on监听
setTimeout(() => {
    eventBus.on('test', (data) => {
        console.log('Received test event:', data);
    });

    eventBus.on('test2', (data) => {
        console.log('Received test2 event:', data);
    });
}, 1000);
